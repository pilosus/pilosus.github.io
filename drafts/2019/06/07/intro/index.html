tags: docker registry

export REGISTRY_LOGIN=pilosus export REGISTRY_PASSWORD=$(pass pilosus/registry) export REGISTRY_URL=docker-registry.pilosus.org export REGISTRY_HEADER=”Accept: application/vnd.docker.distribution.manifest.v2+json”

# Get list of tags for repository "pilosus/pili"
curl -sSL -H "$REGISTRY_HEADER" --user pilosus:$REGISTRY_PASSWORD \
"${REGISTRY_URL}/v2/pilosus/pili/tags/list" | jq '.tags'

# Get digest of "pilosus/pili:pyup-scheduled-update-2019-06-03"
DIGEST=$(curl -sSL -I -H "$REGISTRY_HEADER" --user pilosus:$REGISTRY_PASSWORD \
"${REGISTRY_URL}/v2/pilosus/pili/manifests/pyup-scheduled-update-2019-06-03" \
| awk '$1 == "docker-content-digest:" { print $2 }' | tr -d $'\r')

# Delete image "pilosus/pili:pyup-scheduled-update-2019-06-03"
curl -v -sSL -X DELETE --user pilosus:$REGISTRY_PASSWORD "${REGISTRY_URL}/v2/pilosus/pili/manifests/${DIGEST}"

# garbage collect
kubectl exec -it --namespace <your-registry-ns> <your-docker-registry-pod> \
/bin/registry garbage-collect /etc/docker/registry/config.yml
